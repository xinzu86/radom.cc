      妈妈联盟WeChat App页面

      适用于WebApp、WeChat App等H5移动端页面。

      1. @charset "utf-8";
      2. /*========================================
      3. Version：1.0 Date：2015/06/18
      4. Compatible Kernel：WebKit、Blink、WeChat App
      5. Author：mamalianmeng
      6. ========================================*/
      7. html,body,h1,h2,h3,h4,h5,h6,p,br,form,input,button,textarea,select,fieldset,legend,blockquote,ul,ol,li,dl,dt,dd,pre,table,caption,tr,td,th,article,aside,details,figcaption,figure,footer,header,hgroup,menu,nav,section,summary,time,mark,audio,video{margin:0;padding:0;font:14px "Helvetica";}
      8. html,body{height:100%;-webkit-text-size-adjust:100%;text-size-adjust:100%;}
      9. h1,h2,h3,h4,h5,h6{font-weight:normal;}
      10. button,select,textarea,input{vertical-align:middle;}

      15. i{display:inline-block;}
